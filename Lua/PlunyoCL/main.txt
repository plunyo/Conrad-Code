import "time"
import "string"

local ballPosition = 0
local direction = 1
local screenWidth = 20

while (true) {
    print("
".repeat(screenWidth))

    local index = 0
    while (index < 1) {
        print(" ".repeat(ballPosition) + "O" + " ".repeat(screenWidth - ballPosition - 1))
        index += 1
    }

    ballPosition += direction

    if (ballPosition == screenWidth - 1 or ballPosition == 0) {
        direction *= -1
    } 

    time.sleep(0.1) -- Pause for a moment
}
